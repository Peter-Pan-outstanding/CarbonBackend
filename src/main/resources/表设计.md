### 能效预警

```sql
CREATE TABLE a_energy_alert (
    id INT AUTO_INCREMENT PRIMARY KEY,               -- 编号，自增主键
    monitoring_point_name VARCHAR(100) NOT NULL,     -- 检测点名称
    monitoring_point_address VARCHAR(255) NOT NULL,  -- 检测点地址
    real_time_power DOUBLE,                          -- 实时功率
    real_time_voltage DOUBLE,                        -- 实时电压
    status INT NOT NULL,                             -- 状态 (0: 正常, 1: 异常, 2: 维护)
    report_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP, -- 上报时间，默认为当前时间
    notes TEXT,                                      -- 备注
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,  -- 创建时间，默认为当前时间
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP -- 最后更新时间
);
```

### 项目维护

```sql
CREATE TABLE a_project_maintenance (
    id INT AUTO_INCREMENT PRIMARY KEY,              -- 编号，自增主键
    project_name VARCHAR(100) NOT NULL,             -- 项目名
    manager VARCHAR(100) NOT NULL,                  -- 负责人
    start_date DATE NOT NULL,                       -- 开始日期
    end_date DATE,                                  -- 结束日期 (允许为空)
    status INT NOT NULL,                            -- 项目状态 (0: 进行中, 1: 已完成, 2: 已暂停, 3: 已取消)
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, -- 创建时间，默认当前时间
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP -- 最后更新时间
);
```
